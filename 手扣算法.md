## vector

```c++
#include <iostream>
template <typename T>
class vector{
private:
    T * data;
    size_t _cap;
    size_t _size;
    void setcap(size_t newcap);//内存申请
public:
    vector(int n=1);//构造
    T operator[](const int & x);//赋值运算符重载
    void push_back(const T & x);//添加元素
    T pop_back();//删除末尾元素
    size_t size();//长度
    void clear();//清空
    bool empty();//是否为空
};
template <typename T>
void vector<T>::setcap(size_t newcap){
    T * tmp = new T(newcap);
    for(size_t i = 0; i < _size; ++i){
        *(tmp + i) = *(data + i);
    }
    delete data;
    data = tmp;
    _cap = newcap;
}
//构造函数
template <typename T>
vector<T>::vector(int n):_cap(n){
    this->data = new T[_cap];
    this->_size = 0;
}   
template <typename T>
T vector<T>::operator[](const int & x){
    return *(data + x);
}
template <typename T>
void vector<T>::push_back(const T & x){
    if( _size >= _cap) setcap( _cap * 2 );
    data[_size++] = x;
}
template <typename T>
T vector<T>::pop_back(){
    --_size;
    if(_cap / 4 > _size ) setcap(_cap / 2);
    return *(data + _size);
}
template <typename T>
size_t vector<T>::size(){
    return _size;
}
template <typename T>
void vector<T>::clear(){
    _size = 0;
}
template <typename T>
bool vector<T>::empty(){
    return _size == 0;
}
```



## shared_ptr

shared维护了一个计数器，当计数器为零时才会销毁这段内存。

```c++
template <typename T>
class myshared{
private:
	int * count;
	T * t;
	void add();
	void del();
public :
	myshared(T * x);
	~myshared();
	myshared(const myshared<T>&x);
	myshared<T>& operator=(const myshared<T>&x);
    T& operator *() const;
    T* operator ->() const ;
};
template <typename T>
void myshared<T>::add(){
    if(count == nullptr) count = new int(0);
	++ *count;
}
template <typename T>
void myshared<T>::del(){
	-- *count;
	if(*count == 0){
		delete count;
        count = nullptr;
        delete t;
        t = nullptr;
	}
}
template <typename T>
myshared<T>::myshared(T * x):t(x){
    count = nullptr;
    add();
}
template <typename T>
myshared<T>::~myshared(){
    del();
}
template <typename T>
myshared<T>::myshared(const myshared&x){
    this->count = x.count;
    this->t = x.t;
    add();
}
template <typename T>
myshared<T>& myshared<T>::operator=(const myshared<T> & x){
    del();
    this->count = x.count;
    this-t = t;
    add();
    return *this;
}
template <typename T>
T& myshared<T>::operator*() const{
    return *t;
}
template <typename T>
T* myshared<T>::operator->() const{
    return t;
}
```

## 快排

```c++
template <typename T>
void kuaiPai(std::vector<T> & ve, int st, int en){
    if(st >= en) return;
    int l = st - 1, r = en + 1,ind = ve[l + r >> 1];
    while(l < r){
        do ++l; while(ve[l] < ind);
        do --r; while(ve[r] > ind);
        if(l < r) std::swap(ve[l], ve[r]);
    }
    // 这里一定是r 与 r + 1，因为l可能大于r, 此时使用l + r 违背快排思想。
    kuaiPai(ve, st, r); 
    kuaiPai(ve, r+1, en);
}
```

## 归并

```c++
template <typename T>
void guiBing(std::vector<T> & ve, std::vector<T> & tmp, int l, int r){
    if(l >= r) return ;
    int mid = l + r >> 1;
    guiBing(ve, tmp, l, mid);
    guiBing(ve, tmp, mid+1, r);
    int i = l, j = mid + 1, k = l;
    while(i <= mid && j <= r){
        if(ve[i] < ve[j]) tmp[k++] = ve[i++];
        else tmp[k++] = ve[j++];
    }
    while(j <= r) tmp[k++] = ve[j++];
    while(i <= mid) tmp[k++] = ve[i++];
    while(l <= r) ve[l] = tmp[l++];
}
```

## 堆排

```c++
template <typename T>
class BigHeap{
private:
    std::vector<T> tree;
    void down(int idx){
        while(idx * 2 < tree.size()){
            idx = idx * 2;
            if(idx + 1 < tree.size() && tree[idx] < tree[idx+1]) ++idx;
            if(tree[idx] > tree[idx/2]) std::swap(tree[idx],tree[idx/2]);
            else break;
        }
    }
    void up(int idx){
        while(idx/2 >= 0 && tree[idx/2] < tree[idx]){
            std::swap(tree[idx/2],tree[idx]);
            idx /= 2;
        }
    }
public:
    void add(T & value){
        tree.push_back(value);
        up(tree.size()-1);
    }
    void del(){
        std::swap(tree[0],tree[tree.size()-1]);
        tree.pop_back();
        down(0);
    }
    T get(){
        return tree[0];
    }
};
```
